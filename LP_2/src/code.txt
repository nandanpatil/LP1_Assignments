START 101
ONE DC 1
READ N
MOVER BREG ONE
MOVEM BREG TERM
AGAIN MULT BREG TERM
MOVER CREG TERM
ADD CREG AREG
MOVEM CREG ='5'
X EQU ONE
COMP CREG N
BC LE AGAIN
DIV BREG ='2'
LTORG
X EQU ONE
MOVEM BREG RESULT
MOVEM CREG ='8'
MOVEM CREG ='4'
LTORG
PRINT RESULT
MOVEM CREG ='1'
MOVEM CREG ='1'
STOP
N DS 1
RESULT DS 1
TERM DS 1
TWO DC 2
END
